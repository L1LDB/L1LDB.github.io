def to_36(number):
    global BASE
    
    a = number // 36
    b = number % 36
    
    if not a:
        return BASE[b]
    else:
        return to_36(a) + BASE[b]

BASE = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ"

    NUMBERS.append(input())

    WEIGHT[BASE[i]] = 0
    
for i in range(len(NUMBERS)):
    number = NUMBERS[i]
    
    # ê° ìë¦¿ìˆ˜ ë§Œí¼ ê°€ì¤‘ì¹˜ ì¦ê°€ (ì‘ì€ ìˆ˜ë¥¼ Z ë¡œ ë°”ê¿€ìˆ˜ë¡ ê°’ì€ ë” ì»¤ì§)
    length = len(number);
    for j in range(length):
        digit = number[j]
        WEIGHT[digit] += (35 - int(digit, 36)) * (36 ** (length - j - 1))
    
# ê°€ì¤‘ì¹˜ê°€ ë†’ì€ ìˆœìœ¼ë¡œ ì •ë ¬ í›„ K ê°œ pick

    number = NUMBERS[i]
    current = ""
    for ch in number:
        if ch in picked:
            current = current + "Z"
        else:
            current = current + ch
    sum += int(current, 36)
    
print(to_36(sum))
