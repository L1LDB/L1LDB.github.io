    # μ«μλ¥Ό μ μ™Έν•μ§€ μ•μ„ κ²½μ°
    dp[i][0] = max(dp[i-1][0] + numbers[i], numbers[i])
    # ν•λ‚μ μ«μλ¥Ό μ μ™Έν•  κ²½μ°
    dp[i][1] = max(dp[i-1][0], dp[i-1][1] + numbers[i])

    result = max(result, max(d))
print(result)
