import sys
import copy

input = sys.stdin.readline
r, c, t = map(int, input().split())

graph = []
machine = [] # ê³µê¸°ì²­ì •ê¸° ì¢Œí‘œ

    new_graph = [[0] * c for _ in range(r)]
    for i in range(r):
        for j in range(c):
            if g[i][j] == -1:
                new_graph[i][j] = -1
                continue
            if g[i][j]:
                unit = g[i][j] // 5
                for k in range(4):
                    ny, nx = i + dy[k], j + dx[k]
                    if 0 <= ny < r and 0 <= nx < c and g[ny][nx] != -1:
                        new_graph[ny][nx] += unit
                        g[i][j] -= unit
                new_graph[i][j] += g[i][j]
    return new_graph

    for i in range(4):

        while 0 <= ny < r and 0 <= nx < c and (ny != machine[0] or nx != 0):
            new_graph[ny][nx] = g[y][x]
            y, x = ny, nx
            ny, nx = y + left_y[i], x + left_x[i]

    y, x = machine[1], 0 # ì‹œê³„ ë°©í–¥ ìˆœí™˜(ê³µê¸°ì²­ì •ê¸° ì•„ë˜ìª½ ìˆœí™˜)
    g[y][x] = 0
    for i in range(4):

        while 0 <= ny < r and 0 <= nx < c and (ny != machine[1] or nx != 0):
            new_graph[ny][nx] = g[y][x]
            y, x = ny, nx
            ny, nx = y + right_y[i], x + right_x[i]
    return new_graph


for i in range(t):
    graph = spread(graph)
    graph = clean(graph)

total = 0
for i in range(r):
    for j in range(c):
        total += graph[i][j]
