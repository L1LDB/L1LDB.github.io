```python
def solution(N, number):
    dp = [set([int(str(N)*i)]) for i in range(1, 9)]

    for count in range(8):
        for i in range(count):
            for left in dp[i]:
                for right in dp[count - i - 1]:
                    dp[count].add(left + right)
                    dp[count].add(left - right)
                    dp[count].add(left * right)
                    if right != 0:
                        dp[count].add(left // right)
        if number in dp[count]:
            return count + 1

    return -1
```
