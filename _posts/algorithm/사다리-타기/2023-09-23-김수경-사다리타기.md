import sys

input = sys.stdin.readline
k = int(input())
n = int(input())
fin = list(input().rstrip())
empty_line = 0
ladder = [0] * (k-1)
ans = ''
graph = []
for i in range(n):
    graph.append(list(input().rstrip()))
    if (graph[-1][1] == '?'):
        empty_line = i

# print(empty_line)
for i in range(k):
    now, pin = i, i
    for j in range(n):
        if j == empty_line:
            pin = now
            continue
        if now < k - 1 and graph[j][now] == '-':  # ì˜¤ë¥¸ìª½ìœ¼ë¡œ ê°€ëŠ” ê²½ìš°
            now += 1
        elif now > 0 and graph[j][now - 1] == '-':
            now -= 1
        # print(now)

    # print('í˜„ì¬ ì•ŒíŒŒë²³', ord('A') + i)
    # print('ìµœì¢… ì•ŒíŒŒë²³', ord(fin[now]))
    # print('now', now)
    if ord('A') + i == ord(fin[now]):
        if (pin < k - 1):
            ladder[pin] -= 1
        if (pin > 0):
            ladder[pin - 1] -= 1

    else:
        if (pin < k - 1):
            ladder[pin] += 1
        if (pin > 0):
            ladder[pin - 1] += 1

# print(ladder)
for i in range(k - 1):
    if ladder[i] == 2:
        if i == 0:
            ans += '-'
        elif ans[i - 1] != '-':
            ans += '-'
        elif i < k - 1 and ladder[i + 1] == 2:
            ans += '*'
        else:
            ans = '*' * (k - 1)
            break
    else:
        ans += '*'

print(ans)
