```python
def solution(maps):
    N = len(maps)
    M = len(maps[0])

    delta = [
        (0, 1), (0, -1), (1, 0), (-1, 0)
    ]

    visit = []
    for i in range(N):
        visit.append([0] * M)

    q = []
    q.append((0, 0))

    while q:
        x, y = q.pop(0)

        for dx, dy in delta:
            nx = x + dx
            ny = y + dy

            if 0 <= nx < M and 0 <= ny < N and maps[ny][nx] == 1 and not visit[ny][nx]:
                visit[y][x] = 1
                q.append((nx, ny))
                maps[ny][nx] = maps[y][x] + 1

    if maps[N - 1][M - 1] == 1:
        return -1
    return maps[N - 1][M - 1]
```
