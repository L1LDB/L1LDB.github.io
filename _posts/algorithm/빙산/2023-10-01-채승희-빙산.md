import sys
input = sys.stdin.readline

N, M = map(int, input().split())

    north_pole.append(list(map(int, input().split())))

            iceberg.append((i, j))

dy = [-1, 1, 0, 0]

def bfs(x, y):
    q = deque([])
    q.append((x, y))
    melt_ice = []
    visited[x][y] = True

        minus = 0   # ë°”ë‹¤ ì¸ì ‘ ìˆ˜(ë…¹ëŠ” ì–‘)

        melt_ice.append((x, y, minus))
    return melt_ice 

year = -1

    iceberg = next_iceberg
