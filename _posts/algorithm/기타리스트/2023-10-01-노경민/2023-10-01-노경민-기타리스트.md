N, S, M = map(int, input().split())
volumes = list(map(int, input().split()))

dp = [[0 for j in range(M + 1)] for i in range(N + 1)]

dp[0][S] = 1
for i in range(N):
    for j in range(M + 1):
        if dp[i][j] == 1:
            up = j + volumes[i]
            down = j - volumes[i]
            
            if 0 <= up <= M:
                dp[i + 1][up] = 1
            if 0 <= down <= M:
                dp[i + 1][down] = 1

ans = -1
for i in range(M + 1):
    if dp[N][i] == 1:
        ans = i
print(ans)
