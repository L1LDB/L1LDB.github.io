delta = [[1, 0], [-1, 0], [0, 1], [0, -1]]
R, C = map(int, input().split())

MAP = []
for i in range(R):
    row = input()
    MAP.append(row)

def bfs(x, y):
    visit = [[0] * C for _ in range(R)]
    visit[i][j] = 1
    count = 0
    q = [(x, y)]
    while q:
        cx, cy = q.pop(0)
        for dx, dy in delta:
            nx = cx + dx
            ny = cy + dy
            if 0 <= nx < C and 0 <= ny < R and visit[ny][nx] == 0 and MAP[ny][nx] == 'L':
                visit[ny][nx] = visit[cy][cx] + 1
                count = max(count, visit[ny][nx])
                q.append((nx, ny))
    return count - 1

max_distance = 0
for i in range(R):
    for j in range(C):
        # ëª¨ë“  ìœ¡ì§€ì— ëŒ€í•´ ê²€ì‚¬
